1//! Effective bit-depth estimation.
2//!
3//! Many "24-bit" files are really 16-bit (or less) content padded with zero low
4//! bits — either because the master was 16-bit or because of a lazy conversion.
5//! By OR-ing every integer sample value together, the count of trailing zero
6//! bits (within the declared bit width) reveals how many low bits never carry
7//! information.
8
9/// Given the bitwise OR of every integer sample value and the container's
10/// declared bit width, return the effective (used) bit depth.
11///
12/// The OR mask is restricted to the low `declared_bits` bits so that
13/// sign-extension of negative two's-complement samples does not inflate the
14/// result. `effective = declared_bits - trailing_zero_bits`.
15///
16/// * All-zero input (pure digital silence) is reported as 1 bit.
17pub fn effective_bits(or_mask: u32, declared_bits: u32) -> u32 {
18    let width = declared_bits.clamp(1, 32);
19    let mask = if width >= 32 {
20        u32::MAX
21    } else {
22        (1u32 << width) - 1
23    };
24    let low = or_mask & mask;
25    if low == 0 {
26        return 1;
27    }
28    let trailing = low.trailing_zeros();
29    width.saturating_sub(trailing).max(1)
30}
31
32/// Is a `declared`-bit file effectively 16-bit or less?
33pub fn is_fake_hires(declared: u32, or_mask: u32) -> bool {
34    declared >= 24 && effective_bits(or_mask, declared) <= 16
35}

37#[cfg(test)]
38mod tests {
39    use super::*;
40
41    #[test]
42    fn sixteen_bit_padded_to_24_is_flagged() {
43        // 16-bit samples left-shifted into a 24-bit field => low 8 bits zero.
44        let mut mask = 0u32;
45        for v in [0x1234_00u32, 0x5600_00, 0x00AB_00, 0x7F00_00] {
46            mask |= v;
47        }
48        assert_eq!(effective_bits(mask, 24), 16);
49        assert!(is_fake_hires(24, mask));
50    }
51
52    #[test]
53    fn genuine_24bit_is_not_flagged() {
54        let mut mask = 0u32;
55        for v in [0x123456u32, 0x000001, 0xABCDEF, 0x000003] {
56            mask |= v;
57        }
58        assert!(effective_bits(mask, 24) > 16);
59        assert!(!is_fake_hires(24, mask));
60    }
61
62    #[test]
63    fn sign_extended_negatives_do_not_inflate() {
64        // -256 (0xFFFFFF00) has 8 trailing zero bits; within a 24-bit width the
65        // effective depth is 16, not 32.
66        let mask = (-256i32) as u32;
67        assert_eq!(effective_bits(mask, 24), 16);
68    }
69
70    #[test]
71    fn silence_reports_one_bit() {
72        assert_eq!(effective_bits(0, 24), 1);
73    }
74}
