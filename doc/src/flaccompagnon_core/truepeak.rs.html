1//! True-peak (inter-sample peak) measurement, ITU-R BS.1770-style.
2//!
3//! Digital samples are points on a continuous waveform: between two samples the
4//! reconstructed analog signal can swing *above* both of them. A track whose
5//! sample peak reads −0.2 dBFS can therefore clip a DAC's reconstruction filter
6//! at +1 dBTP. BS.1770-4 measures this by upsampling ×4 with a 48-tap lowpass
7//! FIR and taking the peak of the oversampled stream — exactly what this module
8//! does, as a streaming polyphase filter (12 multiplies per phase, 4 phases per
9//! input sample, per channel).
10//!
11//! The filter is a windowed-sinc (Blackman) lowpass with cutoff at the input
12//! Nyquist, unity gain per phase. Validated against an independent NumPy
13//! implementation: a 0.98 full-scale quarter-rate sine sampled at its
14//! zero-crossing offset reads 0.693 sample peak but 0.9714 true peak
15//! (recovering the hidden crest), and hard-clipped material reads > 1.0
16//! (an inter-sample "over").
17
18/// 4× oversampling factor (BS.1770-4 recommends ≥ 4x for ≤ 96 kHz rates).
19pub const TP_FACTOR: usize = 4;
20/// Taps per polyphase branch (48 total / 4 phases).
21const PHASE_LEN: usize = 12;
22
23/// 48-tap windowed-sinc lowpass, generated and validated in NumPy
24/// (`sinc(n/4) * blackman(48)`, normalized to unity DC gain per phase).
25const TP_TAPS: [f32; 48] = [
26    2.8770393805e-19, -8.4620792140e-05, -3.6092137667e-04, -3.6337902617e-04,
27    7.0568670835e-04, 2.9338012272e-03, 4.6865622501e-03, 2.9440137922e-03,
28    -4.2954750165e-03, -1.4678888264e-02, -2.0272630732e-02, -1.1363293682e-02,
29    1.5119076534e-02, 4.7897022893e-02, 6.2173748295e-02, 3.3175379005e-02,
30    -4.2571243044e-02, -1.3200353510e-01, -1.7082947420e-01, -9.3211746888e-02,
31    1.2718490344e-01, 4.4935166808e-01, 7.7127787555e-01, 9.7258547035e-01,
32    9.7258547035e-01, 7.7127787555e-01, 4.4935166808e-01, 1.2718490344e-01,
33    -9.3211746888e-02, -1.7082947420e-01, -1.3200353510e-01, -4.2571243044e-02,
34    3.3175379005e-02, 6.2173748295e-02, 4.7897022893e-02, 1.5119076534e-02,
35    -1.1363293682e-02, -2.0272630732e-02, -1.4678888264e-02, -4.2954750165e-03,
36    2.9440137922e-03, 4.6865622501e-03, 2.9338012272e-03, 7.0568670835e-04,
37    -3.6337902617e-04, -3.6092137667e-04, -8.4620792140e-05, 2.8770393805e-19,
38];
39
40/// One channel's delay line.
41struct Channel {
42    /// Last `PHASE_LEN` input samples, newest first.
43    delay: [f32; PHASE_LEN],
44}
45
46/// Streaming multi-channel true-peak meter.
47pub struct TruePeak {
48    channels: Vec<Channel>,
49    peak: f32,
50}
51
52impl TruePeak {
53    pub fn new(channels: usize) -> Self {
54        Self {
55            channels: (0..channels.max(1))
56                .map(|_| Channel {
57                    delay: [0.0; PHASE_LEN],
58                })
59                .collect(),
60            peak: 0.0,
61        }
62    }
63
64    /// Feed one frame (one sample per channel).
65    pub fn push_frame(&mut self, samples: &[f32]) {
66        for (ch, &s) in self.channels.iter_mut().zip(samples.iter()) {
67            // Shift the delay line (newest sample at index 0).
68            ch.delay.copy_within(0..PHASE_LEN - 1, 1);
69            ch.delay[0] = s;
70            // Four oversampled outputs: phase p uses taps p, p+4, p+8, ...
71            // With the newest sample first, tap index k*4+p multiplies delay[k].
72            for p in 0..TP_FACTOR {
73                let mut acc = 0.0f32;
74                for k in 0..PHASE_LEN {
75                    acc += TP_TAPS[k * TP_FACTOR + p] * ch.delay[k];
76                }
77                let a = acc.abs();
78                if a > self.peak {
79                    self.peak = a;
80                }
81            }
82        }
83    }
84
85    /// Highest oversampled magnitude seen so far (linear, can exceed 1.0).
86    pub fn peak(&self) -> f32 {
87        self.peak
88    }
89
90    /// True peak in dBTP (0.0 == full scale; positive == inter-sample over).
91    pub fn peak_dbtp(&self) -> f32 {
92        if self.peak > 0.0 {
93            20.0 * self.peak.log10()
94        } else {
95            f32::NEG_INFINITY
96        }
97    }
98}

100#[cfg(test)]
101mod tests {
102    use super::*;
103
104    /// Quarter-rate sine at phase π/4: samples sit at ±0.693 but the true crest
105    /// is 0.98. NumPy ground truth for this filter: 0.97135.
106    #[test]
107    fn recovers_inter_sample_peak() {
108        let fs = 48_000usize;
109        let mut tp = TruePeak::new(1);
110        let mut sample_peak = 0.0f32;
111        for n in 0..fs {
112            // A quarter-rate sine is exactly periodic every 4 samples, so its
113            // phase can be computed from n % 4 instead of the raw n * rate
114            // formula. That matters here: for a full second at 48 kHz the raw
115            // formula's angle argument grows past 75 000 radians, and f32
116            // loses enough precision at that magnitude to visibly perturb
117            // sin() (this was the actual cause of a prior test flake, not the
118            // filter under test). Wrapping keeps the angle small and exact.
119            let phase = std::f32::consts::FRAC_PI_2 * (n % 4) as f32 + std::f32::consts::FRAC_PI_4;
120            let x = 0.98 * phase.sin();
121            sample_peak = sample_peak.max(x.abs());
122            tp.push_frame(&[x]);
123        }
124        assert!(sample_peak < 0.70, "sample peak {sample_peak}");
125        let peak = tp.peak();
126        assert!(
127            (peak - 0.97135).abs() < 0.002,
128            "true peak {peak}, expected ≈0.97135"
129        );
130    }
131
132    /// Hard-clipped sine: reconstruction overshoots full scale (true peak > 1),
133    /// NumPy ground truth 1.01199.
134    #[test]
135    fn clipped_material_reads_over() {
136        let fs = 48_000usize;
137        let mut tp = TruePeak::new(1);
138        for n in 0..fs {
139            let x = (1.4 * (2.0 * std::f32::consts::PI * 997.0 * n as f32 / fs as f32).sin())
140                .clamp(-1.0, 1.0);
141            tp.push_frame(&[x]);
142        }
143        let peak = tp.peak();
144        assert!(
145            (peak - 1.01199).abs() < 0.002,
146            "true peak {peak}, expected ≈1.01199"
147        );
148        assert!(tp.peak_dbtp() > 0.0, "dBTP {} should be positive", tp.peak_dbtp());
149    }
150
151    /// Benign smooth material: true peak stays close to the sample peak
152    /// (no phantom overshoot from the filter itself).
153    #[test]
154    fn benign_material_matches_sample_peak() {
155        let fs = 48_000usize;
156        let mut tp = TruePeak::new(2);
157        let mut sample_peak = 0.0f32;
158        for n in 0..fs {
159            // 100 Hz sine, samples land densely on the crest.
160            let x = 0.5 * (2.0 * std::f32::consts::PI * 100.0 * n as f32 / fs as f32).sin();
161            sample_peak = sample_peak.max(x.abs());
162            tp.push_frame(&[x, x]);
163        }
164        let ratio = tp.peak() / sample_peak;
165        assert!((0.98..=1.05).contains(&ratio), "ratio {ratio}");
166    }
