1//! Authenticity detections, mirroring the three independent tests of the
2//! original *Lossless Audio Checker* (Lacroix & Prime, AES):
3//!
4//! * **Upscaling**   — a low-resolution signal (≤16-bit) stored at a higher bit
5//!   depth. Detected from the effective vs. declared bit depth.
6//! * **Upsampling**  — a low-rate signal placed in a higher-sample-rate
7//!   container; its content is band-limited well below the claimed Nyquist.
8//! * **Transcoding** — a lossy (MP3/AAC) source re-wrapped as lossless. Detected
9//!   from a brick-wall spectral cut-off and, for AAC specifically, from a flat
10//!   high-frequency "dead zone" in the sine-window MDCT domain (the *AAC-SIN*
11//!   signature).
12//!
13//! Each test is independent: a file may trip none, one, or several. As in the
14//! original, these are heuristics — the reasoning is always surfaced in `detail`.
15
16use serde::{Deserialize, Serialize};
17
18use crate::analyzer::AnalysisSummary;

20/// Outcome of the transcoding test.
21#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
22#[serde(rename_all = "snake_case")]
23pub enum TranscodeState {
24    None,
25    Suspected,
26    Detected,
27}
28
29/// The three independent detections plus a human summary.
30#[derive(Debug, Clone, Serialize, Deserialize)]
31pub struct Detections {
32    pub upscaling: bool,
33    pub upsampling: bool,
34    pub transcoding: TranscodeState,
35    /// One-line explanation of the flagged issues (or why the file looks clean).
36    pub detail: String,
37    /// Quick status word: "Clean", "Suspicious", or "Flagged".
38    pub summary: String,
39}
40
41impl Detections {
42    pub fn unknown() -> Self {
43        Detections {
44            upscaling: false,
45            upsampling: false,
46            transcoding: TranscodeState::None,
47            detail: "Not yet analyzed.".to_string(),
48            summary: "Unknown".to_string(),
49        }
50    }
51}

53// --- Tunable thresholds -----------------------------------------------------
54/// At/above this fraction of Nyquist, content is considered full-band.
55const CLEAN_RATIO: f64 = 0.88;
56/// Rates above this are "hi-res" containers subject to the upsampling check.
57const HIRES_RATE: u32 = 48_000;
58/// In a hi-res (> 48 kHz) container, content confined below this frequency
59/// (CD/DVD range, with margin) means the extra bandwidth is empty == upsampled.
60const UPSAMPLE_MAX_HZ: f64 = 30_000.0;
61/// STFT: minimum drop (dB) at the cut-off to call it a brick wall.
62const CLIFF_DB: f32 = 22.0;
63/// STFT: the band above the cut-off must be at least this quiet (dead zone).
64const DEAD_ZONE_DB: f32 = -82.0;
65/// MDCT: fraction of frames that must show a dead zone. Calibrated on real
66/// transcodes: genuine music tops out around 0.30 (dark 1990s metal masters),
67/// while 128–192 kbps AAC sits at ~1.0, so 0.70 gives a safe margin both ways.
68const MDCT_DEAD_FRACTION: f32 = 0.70;
69/// MDCT: mean cut-off must be below this fraction of N to matter.
70const MDCT_CUTOFF_RATIO: f64 = 0.90;
71/// MDCT: the dead zone must be at least this flat/deep (dB rel. frame peak).
72const MDCT_DEAD_DB: f32 = -75.0;
73
74/// Whether the STFT shows a genuine lossy-style brick wall.
75fn stft_brickwall(s: &AnalysisSummary) -> bool {
76    s.cliff_db >= CLIFF_DB && s.above_db < DEAD_ZONE_DB
77}
78
79/// Whether the MDCT shows the AAC-SIN dead-zone signature.
80fn mdct_signature(s: &AnalysisSummary) -> bool {
81    match (s.mdct_dead_fraction, s.mdct_cutoff_ratio, s.mdct_dead_db) {
82        (Some(frac), Some(cutoff), Some(dead)) => {
83            frac >= MDCT_DEAD_FRACTION && cutoff < MDCT_CUTOFF_RATIO && dead < MDCT_DEAD_DB
84        }
85        _ => false,
86    }
87}

89/// Run the three detections.
90pub fn classify(
91    summary: &AnalysisSummary,
92    sample_rate: u32,
93    declared_bits: Option<u32>,
94    real_bit_depth: Option<u32>,
95) -> Detections {
96    let ratio = summary.cutoff_ratio;
97    let brick = stft_brickwall(summary);
98    let mdct_dead = mdct_signature(summary);
99    // The re-quantization grid is near-conclusive evidence of an AAC source
100    // (measured: genuine ≤ 0.014 over all onsets, real transcodes ≥ 0.70).
101    let requant = summary
102        .requant_rate
103        .map(|r| r >= crate::requant::DETECT_RATE)
104        .unwrap_or(false);
105    let mdct_cutoff_hz = summary
106        .mdct_cutoff_ratio
107        .map(|r| r * (sample_rate as f64 / 2.0))
108        .unwrap_or(sample_rate as f64 / 2.0);
109    let mdct_khz = mdct_cutoff_hz / 1000.0;
110    let stft_khz = summary.cutoff_hz / 1000.0;
111
112    // 1. Upscaling — fake resolution: the content uses fewer bits than the
113    //    container declares (the low bits are always zero). Covers 24→16 as well
114    //    as 16→8 and similar.
115    let upscaling = matches!(
116        (declared_bits, real_bit_depth),
117        (Some(d), Some(r)) if r < d
118    );
119
120    // 2. Upsampling — fake sample rate. In a hi-res container (> 48 kHz), a
121    //    consistent MDCT dead zone confined to the CD/DVD range means the extra
122    //    bandwidth is empty. (The STFT cut-off is unreliable on real music — it
123    //    usually reads full-band — so the MDCT dead zone is used here.)
124    let upsampling =
125        sample_rate > HIRES_RATE && mdct_dead && mdct_cutoff_hz <= UPSAMPLE_MAX_HZ;
126
127    // 3. Transcoding — lossy source. Per the LAC paper, transcoding detection
128    //    only applies at <= 48 kHz; above that a band limit is upsampling.
129    //    Evidence strength: re-quantization grid > MDCT dead zone > brick wall.
130    let transcoding = if requant && sample_rate <= HIRES_RATE {
131        TranscodeState::Detected
132    } else if upsampling {
133        TranscodeState::None
134    } else if sample_rate <= HIRES_RATE && mdct_dead {
135        TranscodeState::Detected
136    } else if sample_rate <= HIRES_RATE && brick {
137        TranscodeState::Detected
138    } else if ratio < CLEAN_RATIO {
139        TranscodeState::Suspected
140    } else {
141        TranscodeState::None
142    };
143
144    // Build the human explanation.
145    let mut reasons: Vec<String> = Vec::new();
146    if upscaling {
147        reasons.push(format!(
148            "Upscaling: ~{}-bit content stored as {}-bit",
149            real_bit_depth.unwrap_or(0),
150            declared_bits.unwrap_or(0)
151        ));
152    }
153    if upsampling {
154        reasons.push(format!(
155            "Upsampling: content stops at ~{:.1} kHz in a {:.1} kHz container",
156            mdct_khz,
157            sample_rate as f64 / 1000.0
158        ));
159    }
160    match transcoding {
161        TranscodeState::Detected if requant => reasons.push(format!(
162            "Transcoding: AAC re-quantization grid found at a synchronized MDCT onset ({:.0}% of bands on-grid) — near-conclusive AAC source",
163            summary.requant_rate.unwrap_or(0.0) * 100.0
164        )),
165        TranscodeState::Detected if mdct_dead => reasons.push(format!(
166            "Transcoding: AAC dead zone in the MDCT domain above ~{mdct_khz:.1} kHz (lossy source)"
167        )),
168        TranscodeState::Detected => reasons.push(format!(
169            "Transcoding: brick-wall cut-off at ~{stft_khz:.1} kHz (lossy source)"
170        )),
171        TranscodeState::Suspected => reasons.push(format!(
172            "Possible transcoding: early roll-off at ~{stft_khz:.1} kHz (could be a naturally dark or acoustic master)"
173        )),
174        TranscodeState::None => {}
175    }
176
177    let flagged = upscaling || upsampling || transcoding == TranscodeState::Detected;
178    let summary_word = if flagged {
179        "Flagged"
180    } else if transcoding == TranscodeState::Suspected {
181        "Suspicious"
182    } else {
183        "Clean"
184    };
185
186    let detail = if reasons.is_empty() {
187        format!("Clean — full-band content to ~{stft_khz:.1} kHz, no lossy signature.")
188    } else {
189        reasons.join(" · ")
190    };
191
192    Detections {
193        upscaling,
194        upsampling,
195        transcoding,
196        detail,
197        summary: summary_word.to_string(),
198    }
199}

201#[cfg(test)]
202mod tests {
203    use super::*;
204    use crate::ClippingInfo;
205
206    fn summ(
207        cutoff_hz: f64,
208        sample_rate: u32,
209        cliff_db: f32,
210        above_db: f32,
211        mdct: Option<(f32, f64, f32)>,
212    ) -> AnalysisSummary {
213        let nyq = sample_rate as f64 / 2.0;
214        let (frac, mcr, mdb) = match mdct {
215            Some((f, c, d)) => (Some(f), Some(c), Some(d)),
216            None => (None, None, None),
217        };
218        AnalysisSummary {
219            cutoff_hz,
220            cutoff_ratio: cutoff_hz / nyq,
221            cliff_db,
222            above_db,
223            spectrum_db: vec![],
224            clipping: ClippingInfo {
225                clipped_samples: 0,
226                clip_events: 0,
227                peak: 0.5,
228                peak_dbfs: -6.0,
229                true_peak: 0.5,
230                true_peak_dbtp: -6.0,
231                clipped: false,
232            },
233            fake_stereo: false,
234            real_bit_depth: None,
235            dr_db: None,
236            mdct_cutoff_ratio: mcr,
237            mdct_dead_db: mdb,
238            mdct_dead_fraction: frac,
239            requant_rate: None,
240        }
241    }
242
243    #[test]
244    fn clean_full_band() {
245        let d = classify(&summ(21000.0, 44100, 4.0, -70.0, None), 44100, Some(16), Some(16));
246        assert!(!d.upscaling && !d.upsampling);
247        assert_eq!(d.transcoding, TranscodeState::None);
248        assert_eq!(d.summary, "Clean");
249    }
250
251    #[test]
252    fn brickwall_is_transcoded() {
253        let d = classify(&summ(16000.0, 44100, 40.0, -110.0, None), 44100, Some(16), Some(16));
254        assert_eq!(d.transcoding, TranscodeState::Detected);
255    }
256
257    #[test]
258    fn requant_grid_alone_is_transcoded() {
259        // Full-band spectrum (256 kbps AAC keeps the whole band: no dead zone,
260        // no brick wall) but the re-quantization grid was found.
261        let mut s = summ(21500.0, 44100, 3.0, -70.0, None);
262        s.requant_rate = Some(0.88);
263        let d = classify(&s, 44100, Some(16), Some(16));
264        assert_eq!(d.transcoding, TranscodeState::Detected);
265        assert!(d.detail.contains("re-quantization"));
266    }
267
268    #[test]
269    fn low_requant_rate_does_not_flag() {
270        let mut s = summ(21500.0, 44100, 3.0, -70.0, None);
271        s.requant_rate = Some(0.02); // genuine files measure ≤ ~0.014
272        let d = classify(&s, 44100, Some(16), Some(16));
273        assert_eq!(d.transcoding, TranscodeState::None);
274    }
275
276    #[test]
277    fn aac_mdct_signature_is_transcoded() {
278        // No STFT brick wall, but a clear MDCT dead zone.
279        let d = classify(
280            &summ(19000.0, 44100, 6.0, -70.0, Some((0.85, 0.80, -90.0))),
281            44100,
282            Some(16),
283            Some(16),
284        );
285        assert_eq!(d.transcoding, TranscodeState::Detected);
286    }
287
288    #[test]
289    fn fake_hires_is_upscaled_and_upsampled() {
290        // 96 kHz container, 24-bit declared but 16-bit real, content confined to
291        // ~21.6 kHz (MDCT dead zone) => both upscaled and upsampled.
292        let d = classify(
293            &summ(21600.0, 96000, 40.0, -115.0, Some((1.0, 0.45, -120.0))),
294            96000,
295            Some(24),
296            Some(16),
297        );
298        assert!(d.upscaling);
299        assert!(d.upsampling);
300        assert_eq!(d.transcoding, TranscodeState::None);
301    }
302
303    #[test]
304    fn dark_acoustic_is_suspicious_not_transcoded() {
305        // Early roll-off, gentle (no cliff), no MDCT dead zone.
306        let d = classify(&summ(18000.0, 44100, 8.0, -72.0, Some((0.1, 0.95, -60.0))), 44100, Some(16), Some(16));
307        assert_eq!(d.transcoding, TranscodeState::Suspected);
308        assert_eq!(d.summary, "Suspicious");
309    }
