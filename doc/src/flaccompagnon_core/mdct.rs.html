1//! Modified Discrete Cosine Transform (MDCT) with a sine (Princen–Bradley)
2//! window — the transform AAC uses for its "long" blocks.
3//!
4//! This is used by the AAC-SIN transcoding detector: an AAC encoder quantizes
5//! and zeroes MDCT coefficients (whole scale-factor bands, high frequencies),
6//! leaving a flat, sharply-bounded high-frequency "dead zone" that survives the
7//! decode to PCM. Recomputing the MDCT with AAC's sine window exposes it.
8//!
9//! The transform is implemented directly from its definition (O(N²) per frame).
10//! That is trivially correct — no FFT-twiddle subtleties to get wrong — and the
11//! detector only analyzes a bounded number of frames, so it stays fast. The
12//! cosine basis is built once and shared across all files.
13//!
14//! Definition (input length `2N`, output length `N`):
15//! `X[k] = Σ_{n=0}^{2N-1} w[n]·x[n]·cos[ (π/N)·(n + ½ + N/2)·(k + ½) ]`
16
17use std::sync::OnceLock;
18
19/// AAC long-block size (number of MDCT coefficients).
20pub const AAC_N: usize = 1024;
21
22static SHARED: OnceLock<Mdct> = OnceLock::new();
23
24/// A precomputed MDCT of a fixed size.
25pub struct Mdct {
26    n: usize,
27    /// Sine analysis window, length `2N`.
28    window: Vec<f32>,
29    /// Cosine basis, row-major `[k][t]`, length `N * 2N`.
30    cos: Vec<f32>,
31}
32
33impl Mdct {
34    /// Build an MDCT of output size `n` (input frames are `2n` samples).
35    pub fn new(n: usize) -> Self {
36        let len = 2 * n;
37        let window: Vec<f32> = (0..len)
38            .map(|t| ((std::f32::consts::PI / len as f32) * (t as f32 + 0.5)).sin())
39            .collect();
40        let n0 = n as f32 / 2.0 + 0.5;
41        let mut cos = vec![0.0f32; n * len];
42        let scale = std::f32::consts::PI / n as f32;
43        for k in 0..n {
44            let base = k * len;
45            let kf = k as f32 + 0.5;
46            for t in 0..len {
47                cos[base + t] = (scale * (t as f32 + n0) * kf).cos();
48            }
49        }
50        Self { n, window, cos }
51    }
52
53    /// The shared AAC-sized (`N = 1024`) MDCT, built once on first use.
54    pub fn shared() -> &'static Mdct {
55        SHARED.get_or_init(|| Mdct::new(AAC_N))
56    }
57
58    pub fn n(&self) -> usize {
59        self.n
60    }
61
62    /// Transform one `2N`-sample `frame` into `out` (`N` coefficients).
63    pub fn forward(&self, frame: &[f32], out: &mut [f32]) {
64        let len = 2 * self.n;
65        debug_assert_eq!(frame.len(), len);
66        debug_assert_eq!(out.len(), self.n);
67        for k in 0..self.n {
68            let base = k * len;
69            let mut acc = 0.0f32;
70            for t in 0..len {
71                acc += frame[t] * self.window[t] * self.cos[base + t];
72            }
73            out[k] = acc;
74        }
75    }
76}

78#[cfg(test)]
79mod tests {
80    use super::*;
81
82    #[test]
83    fn tone_energy_concentrates_near_its_bin() {
84        // A mid-band sinusoid at the centre frequency of bin k0 should place the
85        // largest MDCT coefficient at (or adjacent to) k0.
86        let n = 256;
87        let mdct = Mdct::new(n);
88        let len = 2 * n;
89        let k0 = 60usize;
90        // MDCT bin k corresponds to normalized frequency (k + 0.5) / (2N).
91        let f = (k0 as f32 + 0.5) / (len as f32);
92        let frame: Vec<f32> = (0..len)
93            .map(|t| (2.0 * std::f32::consts::PI * f * t as f32).cos())
94            .collect();
95        let mut out = vec![0.0f32; n];
96        mdct.forward(&frame, &mut out);
97
98        let mut argmax = 0usize;
99        let mut max = 0.0f32;
100        for (k, &v) in out.iter().enumerate() {
101            if v.abs() > max {
102                max = v.abs();
103                argmax = k;
104            }
105        }
106        assert!(
107            (argmax as isize - k0 as isize).abs() <= 2,
108            "argmax {argmax} not near {k0}"
109        );
110    }
111
112    #[test]
113    fn silence_transforms_to_zero() {
114        let n = 128;
115        let mdct = Mdct::new(n);
116        let frame = vec![0.0f32; 2 * n];
117        let mut out = vec![9.0f32; n];
118        mdct.forward(&frame, &mut out);
119        assert!(out.iter().all(|&v| v.abs() < 1e-6));
120    }
