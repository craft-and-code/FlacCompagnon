1//! FlacCompagnon core analysis library.
2//!
3//! This crate is intentionally free of any Tauri / UI dependency so that every
4//! analysis routine can be unit-tested in isolation with plain `cargo test`.
5//!
6//! Public entry points:
7//! * [`analyze_file`]  — analyze a single audio file.
8//! * [`analyze_folder`] — analyze every supported audio file inside a folder
9//!   (non-recursive by default; see [`ScanOptions`]).

33/// Audio file extensions FlacCompagnon will attempt to analyze.
34pub const SUPPORTED_EXTENSIONS: &[&str] = &[
35    "flac", "wav", "wave", "aif", "aiff", "aifc", "alac", "m4a", "mp4", "caf", "ogg", "oga",
36    "mp3", "aac", "dsf", "dff",
37];
38
39/// Returns `true` if `path` has an extension FlacCompagnon knows how to decode.
40pub fn is_supported_audio(path: &Path) -> bool {
41    path.extension()
42        .and_then(|e| e.to_str())
43        .map(|e| SUPPORTED_EXTENSIONS.contains(&e.to_ascii_lowercase().as_str()))
44        .unwrap_or(false)
45}

47/// Information about digital clipping found in a file.
48#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
49pub struct ClippingInfo {
50    /// Number of samples at or above the full-scale threshold.
51    pub clipped_samples: u64,
52    /// Number of clip *events* (runs of >= 3 consecutive full-scale samples).
53    pub clip_events: u64,
54    /// Peak absolute sample value observed, normalized to [0, 1].
55    pub peak: f32,
56    /// Peak in dBFS (0.0 == full scale).
57    pub peak_dbfs: f32,
58    /// True peak (inter-sample peak) magnitude from 4x oversampling
59    /// (BS.1770-style). Can exceed 1.0 when the reconstructed waveform
60    /// overshoots full scale between samples.
61    pub true_peak: f32,
62    /// True peak in dBTP. Positive values are inter-sample "overs": the DAC's
63    /// reconstruction filter will clip even though no stored sample does.
64    pub true_peak_dbtp: f32,
65    /// `true` when at least one clip event was detected.
66    pub clipped: bool,
67}

69/// The complete analysis result for one file.
70#[derive(Debug, Clone, Serialize, Deserialize)]
71pub struct FileAnalysis {
72    pub path: String,
73    pub file_name: String,
74    /// Container/codec short name detected from the file's magic bytes
75    /// (e.g. "FLAC", "WAV", "MP4"), independent of the file extension.
76    pub format: String,
77    /// `true` when the real container disagrees with the file extension
78    /// (e.g. a WAV renamed to `.mp3`).
79    pub ext_mismatch: bool,
80    pub sample_rate: u32,
81    pub channels: usize,
82    /// Declared bit depth for integer PCM sources; `None` for float sources.
83    pub declared_bits: Option<u32>,
84    pub duration_secs: f64,
85
86    /// The three LAC-style detections (upscaling / upsampling / transcoding).
87    pub detections: Detections,
88    /// Detected spectral cutoff frequency in Hz.
89    pub cutoff_hz: Option<f64>,
90    /// Cutoff frequency as a ratio of Nyquist (cutoff / (sample_rate/2)).
91    pub cutoff_ratio: Option<f64>,
92
93    /// Estimated *effective* bit depth (integer sources only).
94    pub real_bit_depth: Option<u32>,
95    /// AAC re-quantization hit-rate (0..1); high values prove an AAC source.
96    pub requant_rate: Option<f32>,
97    /// `true` when a >= 2 channel file is actually dual-mono.
98    pub fake_stereo: Option<bool>,
99    /// Verified quality badge: `Some("Hi-Res")` for > 48 kHz or > 16-bit PCM,
100    /// `Some("DSD64")` etc. for DSD — granted only when no detection
101    /// invalidates the claim (no upscaling/upsampling/transcoding).
102    pub badge: Option<String>,
103
104    pub clipping: ClippingInfo,
105
106    /// Dynamic-range estimate in dB (peak vs loudest-20% RMS, DR-meter style).
107    /// High (>= 12 dB) == dynamic master (Full Dynamic Range editions);
108    /// low (< 8 dB) == loudness-war master. `None` when not measurable.
109    pub dr_db: Option<f32>,
110
111    /// FLAC MD5 signature status. `None` for non-FLAC files (no column shown).
112    pub flac_md5: Option<FlacMd5Status>,
113
114    /// Populated when analysis failed; other fields hold best-effort defaults.
115    pub error: Option<String>,
116}

118/// Options controlling how a folder is scanned.
119#[derive(Debug, Clone)]
120pub struct ScanOptions {
121    /// Recurse into sub-folders.
122    pub recursive: bool,
123    /// Compare the FLAC MD5 signature against a hash computed during the single
124    /// decode pass (near-free, since analysis decodes every sample anyway).
125    /// When false, only the signature's presence is reported.
126    pub verify_flac_md5: bool,
127    /// Path to an `ffmpeg` binary, used to decode DSD (DSF/DFF) content.
128    /// `None` limits DSD files to exact header verification.
129    pub ffmpeg: Option<String>,
130}
131
132impl Default for ScanOptions {
133    fn default() -> Self {
134        Self {
135            recursive: true,
136            verify_flac_md5: true,
137            ffmpeg: None,
138        }
139    }
140}

142/// Errors that can occur while analyzing.
143#[derive(Debug, thiserror::Error)]
144pub enum AnalysisError {
145    #[error("io error: {0}")]
146    Io(#[from] std::io::Error),
147    #[error("decode error: {0}")]
148    Decode(String),
149    #[error("unsupported file: {0}")]
150    Unsupported(String),
151}

153/// Analyze a single audio file end-to-end.
154pub fn analyze_file(path: &Path, opts: &ScanOptions) -> FileAnalysis {
155    let file_name = path
156        .file_name()
157        .and_then(|n| n.to_str())
158        .unwrap_or("")
159        .to_string();
160    let path_str = path.to_string_lossy().to_string();
161
162    // Base skeleton so we can early-return a useful record on error.
163    let mut result = FileAnalysis {
164        path: path_str,
165        file_name,
166        format: String::new(),
167        ext_mismatch: false,
168        sample_rate: 0,
169        channels: 0,
170        declared_bits: None,
171        duration_secs: 0.0,
172        detections: Detections::unknown(),
173        cutoff_hz: None,
174        cutoff_ratio: None,
175        real_bit_depth: None,
176        requant_rate: None,
177        fake_stereo: None,
178        badge: None,
179        clipping: ClippingInfo {
180            clipped_samples: 0,
181            clip_events: 0,
182            peak: 0.0,
183            peak_dbfs: f32::NEG_INFINITY,
184            true_peak: 0.0,
185            true_peak_dbtp: f32::NEG_INFINITY,
186            clipped: false,
187        },
188        dr_db: None,
189        flac_md5: None,
190        error: None,
191    };
192
193    let ext = path
194        .extension()
195        .and_then(|e| e.to_str())
196        .map(|e| e.to_ascii_lowercase())
197        .unwrap_or_default();
198    let is_flac = ext == "flac";
199
200    // DSD files take a dedicated path: exact header verification, then (when
201    // ffmpeg is available) content analysis on the decoded PCM.
202    if ext == "dsf" || ext == "dff" {
203        analyze_dsd(path, opts, &mut result);
204        if let Some(detected) = decode::detect_container(path) {
205            if let Some(expected) = decode::ext_canonical(path) {
206                result.ext_mismatch = detected != expected;
207            }
208        }
209        return result;
210    }
211
212    // Single decode pass. FLAC files go through the fused claxon path, which
213    // feeds the analyzer AND hashes the MD5 from the same decoded samples —
214    // nothing is decoded twice. If that fast path fails (corrupt or misnamed
215    // file), fall back to symphonia so the analysis still happens; the MD5
216    // column then reports the error.
217    let mut flac_md5_status: Option<FlacMd5Status> = None;
218    // Sigma-delta noise heritage of a DSD master, when found in hi-res PCM.
219    let mut dsd_heritage: Option<f32> = None;
220    let decoded = if is_flac {
221        match decode::decode_and_analyze_flac(path, opts.verify_flac_md5) {
222            Ok((outcome, md5)) => {
223                flac_md5_status = Some(md5);
224                Ok(outcome)
225            }
226            Err(e) => {
227                flac_md5_status = Some(FlacMd5Status::Error(e.to_string()));
228                decode::decode_and_analyze(path)
229            }
230        }
231    } else {
232        decode::decode_and_analyze(path)
233    };
234
235    match decoded {
236        Ok(outcome) => {
237            result.format = outcome.format;
238            result.sample_rate = outcome.sample_rate;
239            result.channels = outcome.channels;
240            result.declared_bits = outcome.declared_bits;
241            result.duration_secs = outcome.duration_secs;
242
243            let summary = outcome
244                .analyzer
245                .finish(outcome.sample_rate, outcome.declared_bits);
246
247            result.cutoff_hz = Some(summary.cutoff_hz);
248            result.cutoff_ratio = Some(summary.cutoff_ratio);
249            result.requant_rate = summary.requant_rate;
250            result.clipping = summary.clipping.clone();
251            result.dr_db = summary.dr_db;
252            dsd_heritage = dsd::dsd_heritage_check(
253                &summary.spectrum_db,
254                outcome.sample_rate,
255                summary.spectrum_db.len().saturating_sub(1) * 2,
256            );
257
258            if outcome.channels >= 2 {
259                result.fake_stereo = Some(summary.fake_stereo);
260            }
261            let real_bits = match (outcome.declared_bits, summary.real_bit_depth) {
262                (Some(_), Some(real)) => {
263                    result.real_bit_depth = Some(real);
264                    Some(real)
265                }
266                _ => None,
267            };
268            result.detections = detections::classify(
269                &summary,
270                outcome.sample_rate,
271                outcome.declared_bits,
272                real_bits,
273            );
274        }
275        Err(e) => {
276            result.error = Some(e.to_string());
277        }
278    }
279
280    // Real container from magic bytes; flag a mismatch with the extension.
281    if let Some(detected) = decode::detect_container(path) {
282        result.format = detected.to_string();
283        if let Some(expected) = decode::ext_canonical(path) {
284            result.ext_mismatch = detected != expected;
285        }
286    }
287
288    // FLAC MD5 signature, computed during the single decode pass above.
289    if is_flac {
290        result.flac_md5 = flac_md5_status;
291    }
292
293    // Verified Hi-Res badge: hi-res specs that no detection contradicts.
294    let hires_specs = result.sample_rate > 48_000 || result.declared_bits.map_or(false, |b| b > 16);
295    if hires_specs
296        && result.error.is_none()
297        && !result.detections.upscaling
298        && !result.detections.upsampling
299        && result.detections.transcoding != TranscodeState::Detected
300    {
301        result.badge = Some(if dsd_heritage.is_some() {
302            // Hi-res PCM carrying the sigma-delta noise signature of a DSD master.
303            "Hi-Res (DSD source)".to_string()
304        } else {
305            "Hi-Res".to_string()
306        });
307    }
308
309    result
310}

312/// DSD analysis: exact DSF/DFF header verification plus, when ffmpeg is
313/// available, a content check on the decoded PCM (PCM-source brick wall).
314fn analyze_dsd(path: &Path, opts: &ScanOptions, result: &mut FileAnalysis) {
315    let info = match dsd::parse(path) {
316        Ok(i) => i,
317        Err(e) => {
318            result.error = Some(e.to_string());
319            return;
320        }
321    };
322    result.format = info.label();
323    result.sample_rate = info.sample_rate;
324    result.channels = info.channels;
325    result.duration_secs = info.duration_secs();
326
327    let mut flagged: Option<dsd::PcmSourceCheck> = None;
328    let mut analyzed = false;
329
330    if info.dst_compressed {
331        result.detections = Detections {
332            upscaling: false,
333            upsampling: false,
334            transcoding: TranscodeState::None,
335            detail: "DST-compressed DFF: header verified; content analysis is not supported for DST streams.".to_string(),
336            summary: "Unknown".to_string(),
337        };
338    } else if let Some(ffmpeg) = &opts.ffmpeg {
339        let decoded_rate = (info.sample_rate / 8).max(1);
340        match decode::decode_and_analyze_dsd(ffmpeg, path, info.channels, decoded_rate) {
341            Ok(outcome) => {
342                analyzed = true;
343                if result.duration_secs == 0.0 {
344                    result.duration_secs = outcome.duration_secs;
345                }
346                let summary = outcome.analyzer.finish(decoded_rate, None);
347                result.cutoff_hz = Some(summary.cutoff_hz);
348                result.cutoff_ratio = Some(summary.cutoff_ratio);
349                result.clipping = summary.clipping.clone();
350                result.dr_db = summary.dr_db;
351                if info.channels >= 2 {
352                    result.fake_stereo = Some(summary.fake_stereo);
353                }
354                let fft_size = summary.spectrum_db.len().saturating_sub(1) * 2;
355                flagged = dsd::pcm_source_check(&summary.spectrum_db, decoded_rate, fft_size);
356                result.detections = match flagged {
357                    Some(hit) => Detections {
358                        upscaling: false,
359                        upsampling: true,
360                        transcoding: TranscodeState::None,
361                        detail: format!(
362                            "Upsampling: PCM-sourced DSD — digital brick wall at ~{:.2} kHz ({:.0} dB drop). The 1-bit stream was converted from a {} PCM source.",
363                            hit.boundary_hz / 1000.0,
364                            hit.drop_db,
365                            if hit.boundary_hz < 23_000.0 { "44.1 kHz" } else { "48 kHz" }
366                        ),
367                        summary: "Flagged".to_string(),
368                    },
369                    None => Detections {
370                        upscaling: false,
371                        upsampling: false,
372                        transcoding: TranscodeState::None,
373                        detail: "Content blends into the sigma-delta noise shaping with no PCM brick wall — consistent with native DSD.".to_string(),
374                        summary: "Clean".to_string(),
375                    },
376                };
377            }
378            Err(e) => {
379                result.error = Some(e.to_string());
380            }
381        }
382    } else {
383        result.detections = Detections {
384            upscaling: false,
385            upsampling: false,
386            transcoding: TranscodeState::None,
387            detail: "DSD header verified. Install ffmpeg to enable the content authenticity check.".to_string(),
388            summary: "Unknown".to_string(),
389        };
390    }
391
392    // DSD badge: header authentic and content not flagged as PCM-sourced.
393    if result.error.is_none() && flagged.is_none() {
394        result.badge = Some(if analyzed {
395            info.label()
396        } else {
397            format!("{} (unverified)", info.label())
398        });
399    }
400}

402/// A single analyzed folder together with the files it contains.

404pub struct FolderReport {
405    pub root: String,
406    pub files: Vec<FileAnalysis>,
407    /// `true` if any FLAC files were present (the UI shows the MD5 column then).
408    pub has_flac: bool,
409}

411/// List every supported audio file under `root`, sorted, skipping any file that
412/// lives inside a generated `spectres` folder.
413pub fn list_audio_files(root: &Path, recursive: bool) -> Vec<PathBuf> {
414    let mut paths: Vec<PathBuf> = Vec::new();
415    let walker = walkdir::WalkDir::new(root).max_depth(if recursive { usize::MAX } else { 1 });
416    for entry in walker.into_iter().filter_map(|e| e.ok()) {
417        let p = entry.path();
418        if p.is_file() && is_supported_audio(p) {
419            if p.components().any(|c| c.as_os_str() == "spectres") {
420                continue;
421            }
422            paths.push(p.to_path_buf());
423        }
424    }
425    paths.sort();
426    paths
427}
428
429/// Analyze every supported audio file under `root`.
430pub fn analyze_folder(root: &Path, opts: &ScanOptions) -> Result<FolderReport, AnalysisError> {
431    let paths = list_audio_files(root, opts.recursive);
432    let files: Vec<FileAnalysis> = paths.iter().map(|p| analyze_file(p, opts)).collect();
433    let has_flac = files.iter().any(|f| f.flac_md5.is_some());
434
435    Ok(FolderReport {
436        root: root.to_string_lossy().to_string(),
437        files,
438        has_flac,
439    })
