1//! Streaming analyzer.
2//!
3//! The decoder feeds audio one frame at a time (a frame = one sample per
4//! channel). Storing an entire album in memory would be wasteful, so every
5//! metric is accumulated incrementally:
6//!
7//! * spectrum      -> Hann-windowed FFT of a mono downmix, averaged over windows
8//! * clipping      -> full-scale sample counting with run detection
9//! * fake stereo   -> energy of the L-R difference vs. the signal energy
10//! * real bitdepth -> bitwise OR of every integer sample value
11//!
12//! Nothing here depends on a specific file format; [`decode`](crate::decode)
13//! adapts each codec to the [`StreamAnalyzer::push_frame`] interface.
14
15use std::sync::Arc;
16
17use rustfft::{num_complex::Complex, Fft, FftPlanner};
18
19use crate::mdct::{Mdct, AAC_N};
20use crate::{bitdepth, clipping, requant, spectrum, ClippingInfo};

22/// Full-scale detection threshold (normalized). Samples with |value| at or
23/// above this are treated as clipped.
24const CLIP_THRESHOLD: f32 = 0.9997;
25/// FFT window size. 8192 gives ~5.4 Hz resolution at 44.1 kHz.
26const FFT_SIZE: usize = 8192;
27
28// --- MDCT (AAC-SIN transcode detector) -------------------------------------
29/// Analyze one in every `MDCT_STRIDE` overlapping MDCT hops, so the transform
30/// samples the whole track without processing every frame (O(N²) per frame).
31const MDCT_STRIDE: u64 = 4;
32/// Cap on analyzed MDCT frames — keeps a long album fast.
33const MDCT_MAX_FRAMES: u32 = 240;
34/// A dead zone only "exists" if the per-frame cutoff is below this fraction of N.
35const MDCT_DEAD_TOP_RATIO: f32 = 0.92;
36
37// --- Re-quantization detector segment buffers -------------------------------
38/// Preferred segment start (sample index; multiple of 1024 ≈ 8 s at 44.1 kHz).
39const REQ_MAIN_START: u64 = 344 * 1024;
40
41// --- Dynamics (DR) ----------------------------------------------------------
42/// Frames per RMS block (~3 s at 44.1 kHz). The exact wall-clock length is not
43/// critical: the DR estimate averages the loudest blocks, so the granularity
44/// only needs to be long enough to smooth individual drum hits.
45const DYN_BLOCK_FRAMES: u64 = 131_072;
46/// Fraction of the loudest blocks that defines the "loud passages" RMS,
47/// mirroring the classic DR-meter approach (peak vs loudest-20% RMS).
48const DYN_TOP_FRACTION: f64 = 0.2;
49
50/// Aggregated results produced by [`StreamAnalyzer::finish`].
51#[derive(Debug, Clone)]
52pub struct AnalysisSummary {
53    pub cutoff_hz: f64,
54    pub cutoff_ratio: f64,
55    /// How sharply the level drops at the cutoff (dB). Large == a brick wall.
56    pub cliff_db: f32,
57    /// Mean level just above the cutoff (dB rel. peak) — the dead-zone depth.
58    pub above_db: f32,
59    /// Averaged magnitude spectrum in dB, one entry per FFT bin (0..=N/2).
60    pub spectrum_db: Vec<f32>,
61    pub clipping: ClippingInfo,
62    pub fake_stereo: bool,
63    pub real_bit_depth: Option<u32>,
64
65    /// Dynamic-range estimate in dB: peak level vs the RMS of the loudest 20%
66    /// of ~3 s blocks (crest factor of the loud passages, DR-meter style).
67    /// High values (>= 12 dB) indicate a dynamic master; low values (< 8 dB)
68    /// a loudness-war master. `None` for silent or extremely short streams.
69    pub dr_db: Option<f32>,
70
71    // --- MDCT (AAC-SIN) transcode evidence ---
72    /// Mean per-frame MDCT cutoff as a fraction of Nyquist (dead-zone frames).
73    pub mdct_cutoff_ratio: Option<f64>,
74    /// Mean level (dB rel. frame peak) of the MDCT dead zone above the cutoff.
75    pub mdct_dead_db: Option<f32>,
76    /// Fraction of analyzed MDCT frames that showed a high-frequency dead zone.
77    pub mdct_dead_fraction: Option<f32>,
78
79    // --- AAC re-quantization evidence (strongest transcode proof) ---
80    /// Hit-rate of on-grid bands at the best synchronized MDCT onset (0..1).
81    /// ≥ [`requant::DETECT_RATE`] means an AAC source. `None` when the check
82    /// did not run (unsupported rate or file too short).
83    pub requant_rate: Option<f32>,
84}

86/// Incremental audio analyzer.
87pub struct StreamAnalyzer {
88    channels: usize,
89
90    // --- spectrum ---
91    fft: Arc<dyn Fft<f32>>,
92    hann: Vec<f32>,
93    frame_buf: Vec<f32>, // mono accumulation buffer, length grows to FFT_SIZE
94    power_acc: Vec<f64>, // accumulated |X|^2 per bin, length FFT_SIZE/2 + 1
95    window_count: u64,
96
97    // --- clipping ---
98    clip_state: clipping::ClipState,
99
100    // --- fake stereo ---
101    diff_energy: f64,
102    l_energy: f64,
103    r_energy: f64,
104    identical_frames: u64,
105    total_frames: u64,
106
107    // --- bit depth ---
108    int_or_mask: u32,
109    saw_integer: bool,
110
111    // --- MDCT (AAC-SIN) ---
112    mdct: &'static Mdct,
113    mdct_prev: Vec<f32>, // previous hop (N mono samples)
114    mdct_fill: Vec<f32>, // current hop being filled
115    mdct_have_prev: bool,
116    mdct_hop: u64,
117    mdct_scratch: Vec<f32>, // N coefficients
118    mdct_frames: u32,
119    mdct_dead_frames: u32,
120    mdct_cutoff_ratio_sum: f64,
121    mdct_dead_db_sum: f64,
122
123    // --- Re-quantization segment buffers (L/R as f64) ---
124    req_early_l: Vec<f64>,
125    req_early_r: Vec<f64>,
126    req_main_l: Vec<f64>,
127    req_main_r: Vec<f64>,
128
129    // --- dynamics (DR) ---
130    dyn_block_sumsq: f64, // running sum of per-frame mean squares
131    dyn_block_frames: u64,
132    dyn_blocks: Vec<f64>, // mean square of each completed block
133}
134
135impl StreamAnalyzer {
136    pub fn new(channels: usize) -> Self {
137        let mut planner = FftPlanner::<f32>::new();
138        let fft = planner.plan_fft_forward(FFT_SIZE);
139        let hann: Vec<f32> = (0..FFT_SIZE)
140            .map(|n| {
141                let x = std::f32::consts::PI * n as f32 / (FFT_SIZE as f32 - 1.0);
142                x.sin().powi(2) // Hann window == sin^2
143            })
144            .collect();
145        Self {
146            channels: channels.max(1),
147            fft,
148            hann,
149            frame_buf: Vec::with_capacity(FFT_SIZE),
150            power_acc: vec![0.0; FFT_SIZE / 2 + 1],
151            window_count: 0,
152            clip_state: clipping::ClipState::new(CLIP_THRESHOLD),
153            diff_energy: 0.0,
154            l_energy: 0.0,
155            r_energy: 0.0,
156            identical_frames: 0,
157            total_frames: 0,
158            int_or_mask: 0,
159            saw_integer: false,
160            mdct: Mdct::shared(),
161            mdct_prev: Vec::with_capacity(AAC_N),
162            mdct_fill: Vec::with_capacity(AAC_N),
163            mdct_have_prev: false,
164            mdct_hop: 0,
165            mdct_scratch: vec![0.0; AAC_N],
166            mdct_frames: 0,
167            mdct_dead_frames: 0,
168            mdct_cutoff_ratio_sum: 0.0,
169            mdct_dead_db_sum: 0.0,
170            req_early_l: Vec::with_capacity(requant::SEGMENT_LEN),
171            req_early_r: Vec::with_capacity(requant::SEGMENT_LEN),
172            req_main_l: Vec::with_capacity(requant::SEGMENT_LEN),
173            req_main_r: Vec::with_capacity(requant::SEGMENT_LEN),
174            dyn_block_sumsq: 0.0,
175            dyn_block_frames: 0,
176            dyn_blocks: Vec::new(),
177        }
178    }
179
180    /// Push one frame of normalized-float samples (`samples.len() == channels`),
181    /// optionally accompanied by the raw integer sample values for the same
182    /// frame (used for effective bit-depth estimation).
183    pub fn push_frame(&mut self, samples: &[f32], int_samples: Option<&[i32]>) {
184        if samples.is_empty() {
185            return;
186        }
187        let idx = self.total_frames; // sample index of this frame
188        self.total_frames += 1;
189
190        // Capture the re-quantization segments (first two channels, f64).
191        // Both segment starts are multiples of 1024, preserving AAC frame
192        // alignment modulo the block length.
193        let l = samples[0] as f64;
194        let r = if samples.len() >= 2 { samples[1] as f64 } else { l };
195        if self.req_early_l.len() < requant::SEGMENT_LEN {
196            self.req_early_l.push(l);
197            if self.channels >= 2 {
198                self.req_early_r.push(r);
199            }
200        }
201        if idx >= REQ_MAIN_START && self.req_main_l.len() < requant::SEGMENT_LEN {
202            self.req_main_l.push(l);
203            if self.channels >= 2 {
204                self.req_main_r.push(r);
205            }
206        }
207
208        // Clipping + peak (per channel), and the frame's mean square for the
209        // dynamics blocks.
210        let mut frame_sumsq = 0.0f64;
211        for &s in samples {
212            self.clip_state.push(s);
213            frame_sumsq += (s as f64) * (s as f64);
214        }
215        self.dyn_block_sumsq += frame_sumsq / samples.len() as f64;
216        self.dyn_block_frames += 1;
217        if self.dyn_block_frames == DYN_BLOCK_FRAMES {
218            self.dyn_blocks
219                .push(self.dyn_block_sumsq / self.dyn_block_frames as f64);
220            self.dyn_block_sumsq = 0.0;
221            self.dyn_block_frames = 0;
222        }
223
224        // Stereo difference energy (first two channels).
225        if self.channels >= 2 && samples.len() >= 2 {
226            let l = samples[0] as f64;
227            let r = samples[1] as f64;
228            let d = l - r;
229            self.diff_energy += d * d;
230            self.l_energy += l * l;
231            self.r_energy += r * r;
232            if (l - r).abs() < 1e-9 {
233                self.identical_frames += 1;
234            }
235        }
236
237        // Integer OR mask for bit-depth estimation. Raw two's-complement value;
238        // sign extension is handled later by masking to the declared width.
239        if let Some(ints) = int_samples {
240            self.saw_integer = true;
241            for &v in ints {
242                self.int_or_mask |= v as u32;
243            }
244        }
245
246        // Mono downmix into the FFT buffer.
247        let mut mono = 0.0f32;
248        for &s in samples {
249            mono += s;
250        }
251        mono /= samples.len().max(1) as f32;
252        self.frame_buf.push(mono);
253        if self.frame_buf.len() == FFT_SIZE {
254            self.process_window();
255            self.frame_buf.clear();
256        }
257
258        // Feed the same mono sample to the MDCT pipeline (frame = 2N, hop = N).
259        self.mdct_fill.push(mono);
260        if self.mdct_fill.len() == AAC_N {
261            self.mdct_hop += 1;
262            if self.mdct_have_prev
263                && self.mdct_frames < MDCT_MAX_FRAMES
264                && self.mdct_hop % MDCT_STRIDE == 0
265            {
266                self.process_mdct();
267            }
268            std::mem::swap(&mut self.mdct_prev, &mut self.mdct_fill);
269            self.mdct_fill.clear();
270            self.mdct_have_prev = true;
271        }
272    }
273
274    /// Analyze one overlapping MDCT frame (previous hop + current hop) for the
275    /// AAC-SIN transcode signature: a flat, sharply-bounded high-frequency dead
276    /// zone in the sine-window MDCT domain.
277    fn process_mdct(&mut self) {
278        let n = AAC_N;
279        let mut frame = Vec::with_capacity(2 * n);
280        frame.extend_from_slice(&self.mdct_prev);
281        frame.extend_from_slice(&self.mdct_fill);
282
283        let m = self.mdct; // &'static, cheap to copy
284        m.forward(&frame, &mut self.mdct_scratch);
285
286        let mut peak = 0.0f32;
287        for &c in &self.mdct_scratch {
288            let a = c.abs();
289            if a > peak {
290                peak = a;
291            }
292        }
293        self.mdct_frames += 1;
294        if peak < 1e-7 {
295            return; // silent frame
296        }
297
298        let thr = peak * 1e-4; // -80 dB relative to the frame peak
299        let mut cutoff = 0usize;
300        for k in (0..n).rev() {
301            if self.mdct_scratch[k].abs() > thr {
302                cutoff = k;
303                break;
304            }
305        }
306
307        let top = (MDCT_DEAD_TOP_RATIO * n as f32) as usize;
308        if cutoff < top {
309            let start = (cutoff + 8).min(n);
310            if start < n {
311                let mut sum = 0.0f64;
312                let mut cnt = 0u32;
313                for k in start..n {
314                    let d = 20.0 * (self.mdct_scratch[k].abs() / peak).max(1e-12).log10();
315                    sum += d as f64;
316                    cnt += 1;
317                }
318                if cnt > 0 {
319                    self.mdct_dead_db_sum += sum / cnt as f64;
320                    self.mdct_cutoff_ratio_sum += cutoff as f64 / n as f64;
321                    self.mdct_dead_frames += 1;
322                }
323            }
324        }
325    }
326
327    fn process_window(&mut self) {
328        let mut buf: Vec<Complex<f32>> = self
329            .frame_buf
330            .iter()
331            .zip(self.hann.iter())
332            .map(|(&s, &w)| Complex {
333                re: s * w,
334                im: 0.0,
335            })
336            .collect();
337        self.fft.process(&mut buf);
338        for (bin, c) in buf.iter().take(self.power_acc.len()).enumerate() {
339            self.power_acc[bin] += (c.re as f64).powi(2) + (c.im as f64).powi(2);
340        }
341        self.window_count += 1;
342    }
343
344    /// Finalize and compute all summary metrics for a stream at `sample_rate`.
345    /// `declared_bits` is the container's stated integer bit depth (if any) and
346    /// is used to bound the effective bit-depth estimate.
347    pub fn finish(mut self, sample_rate: u32, declared_bits: Option<u32>) -> AnalysisSummary {
348        // Flush a trailing partial window (zero padded) so short files still
349        // produce a spectrum.
350        if self.window_count == 0 && !self.frame_buf.is_empty() {
351            self.frame_buf.resize(FFT_SIZE, 0.0);
352            self.process_window();
353        }
354
355        let spectrum_db = spectrum::average_to_db(&self.power_acc, self.window_count);
356        let (cutoff_hz, cutoff_ratio) =
357            spectrum::detect_cutoff(&spectrum_db, sample_rate, FFT_SIZE);
358        let (cliff_db, above_db) =
359            spectrum::cutoff_context(&spectrum_db, sample_rate, FFT_SIZE, cutoff_hz);
360
361        // Dynamics: flush a meaningful trailing partial block (>= 1/4 length),
362        // then compare the peak with the RMS of the loudest 20% of blocks.
363        if self.dyn_block_frames >= DYN_BLOCK_FRAMES / 4 {
364            self.dyn_blocks
365                .push(self.dyn_block_sumsq / self.dyn_block_frames as f64);
366        }
367        let clipping = self.clip_state.finish(self.channels, self.total_frames);
368        let dr_db = {
369            let mut blocks = self.dyn_blocks.clone();
370            blocks.sort_by(|a, b| b.partial_cmp(a).unwrap_or(std::cmp::Ordering::Equal));
371            let top = ((blocks.len() as f64 * DYN_TOP_FRACTION).ceil() as usize).max(1);
372            let loud: Vec<f64> = blocks.into_iter().take(top).collect();
373            if loud.is_empty() {
374                None
375            } else {
376                let rms = (loud.iter().sum::<f64>() / loud.len() as f64).sqrt();
377                if rms > 1e-9 && clipping.peak > 0.0 {
378                    Some((20.0 * (clipping.peak as f64 / rms).log10()) as f32)
379                } else {
380                    None
381                }
382            }
383        };
384
385        let fake_stereo = if self.channels >= 2 {
386            crate::stereo::is_fake(
387                self.diff_energy,
388                self.l_energy,
389                self.r_energy,
390                self.identical_frames,
391                self.total_frames,
392            )
393        } else {
394            false
395        };
396
397        let real_bit_depth = match (self.saw_integer, declared_bits) {
398            (true, Some(declared)) => Some(bitdepth::effective_bits(self.int_or_mask, declared)),
399            _ => None,
400        };
401
402        // AAC re-quantization check — only meaningful at the AAC frame rates
403        // covered by the 44.1/48 kHz scale-factor band table.
404        let requant_rate = if sample_rate == 44_100 || sample_rate == 48_000 {
405            let (seg_l, seg_r) = if self.req_main_l.len() >= requant::SEGMENT_LEN {
406                (&self.req_main_l, &self.req_main_r)
407            } else {
408                (&self.req_early_l, &self.req_early_r)
409            };
410            let right = if self.channels >= 2 && seg_r.len() >= requant::SEGMENT_LEN {
411                Some(seg_r.as_slice())
412            } else {
413                None
414            };
415            requant::analyze_segment(seg_l, right).map(|r| r.rate)
416        } else {
417            None
418        };
419
420        let (mdct_cutoff_ratio, mdct_dead_db, mdct_dead_fraction) = if self.mdct_frames > 0 {
421            let frac = self.mdct_dead_frames as f32 / self.mdct_frames as f32;
422            if self.mdct_dead_frames > 0 {
423                (
424                    Some(self.mdct_cutoff_ratio_sum / self.mdct_dead_frames as f64),
425                    Some((self.mdct_dead_db_sum / self.mdct_dead_frames as f64) as f32),
426                    Some(frac),
427                )
428            } else {
429                (None, None, Some(frac))
430            }
431        } else {
432            (None, None, None)
433        };
434
435        AnalysisSummary {
436            cutoff_hz,
437            cutoff_ratio,
438            cliff_db,
439            above_db,
440            spectrum_db,
441            clipping,
442            fake_stereo,
443            real_bit_depth,
444            dr_db,
445            mdct_cutoff_ratio,
446            mdct_dead_db,
447            mdct_dead_fraction,
448            requant_rate,
449        }
450    }
451}

453#[cfg(test)]
454mod tests {
455    use super::*;

457    /// A pure sine has a crest factor of exactly sqrt(2) == 3.01 dB; the DR
458    /// estimate on a steady full-scale sine must land there.
459    #[test]
460    fn dr_of_a_pure_sine_is_3db() {
461        let mut a = StreamAnalyzer::new(1);
462        let rate = 44_100.0f64;
463        for n in 0..(DYN_BLOCK_FRAMES * 2) {
464            let s = (2.0 * std::f64::consts::PI * 1000.0 * n as f64 / rate).sin() as f32 * 0.9;
465            a.push_frame(&[s], None);
466        }
467        let summary = a.finish(44_100, None);
468        let dr = summary.dr_db.expect("dr computed");
469        assert!((dr - 3.01).abs() < 0.1, "dr = {dr}");
470    }
471
472    /// A hard-clipped ("brickwalled") sine approaches a square wave whose
473    /// crest factor tends to 0 dB — the loudness-war signature.
474    #[test]
475    fn dr_of_a_squashed_sine_is_low() {
476        let mut a = StreamAnalyzer::new(1);
477        let rate = 44_100.0f64;
478        for n in 0..(DYN_BLOCK_FRAMES * 2) {
479            let raw = (2.0 * std::f64::consts::PI * 1000.0 * n as f64 / rate).sin() * 8.0;
480            let s = raw.clamp(-0.98, 0.98) as f32;
481            a.push_frame(&[s], None);
482        }
483        let summary = a.finish(44_100, None);
484        let dr = summary.dr_db.expect("dr computed");
485        assert!(dr < 1.0, "dr = {dr}");
486    }
487
488    /// Silence yields no DR value rather than a bogus number.
489    #[test]
490    fn dr_of_silence_is_none() {
491        let mut a = StreamAnalyzer::new(1);
492        for _ in 0..(DYN_BLOCK_FRAMES + 10) {
493            a.push_frame(&[0.0], None);
494        }
495        let summary = a.finish(44_100, None);
496        assert!(summary.dr_db.is_none());
497    }
